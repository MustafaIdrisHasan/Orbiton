CREATE EXTENSION IF NOT EXISTS pgcrypto; CREATE TABLE IF NOT EXISTS users ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), email VARCHAR(255) NOT NULL UNIQUE, password_hash_txt TEXT NOT NULL, ph_no TEXT NOT NULL, is_active BOOLEAN DEFAULT TRUE, is_email_verified BOOLEAN DEFAULT FALSE, mfa_enabled BOOLEAN DEFAULT FALSE, mfa_secret TEXT, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ); CREATE TABLE IF NOT EXISTS roles ( id INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY, name VARCHAR(50) NOT NULL UNIQUE, description TEXT ); CREATE TABLE IF NOT EXISTS skills ( id INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY, name VARCHAR(100) NOT NULL UNIQUE, category VARCHAR(100) ); CREATE TABLE IF NOT EXISTS student_profiles ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), user_id UUID UNIQUE REFERENCES users(id) ON DELETE CASCADE, full_name VARCHAR(255) NOT NULL, department VARCHAR(100), program VARCHAR(100), year_of_study INTEGER, cgpa NUMERIC(4,2), backlog_count INTEGER DEFAULT 0, date_of_birth DATE, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ); CREATE TABLE IF NOT EXISTS recruiter_profiles ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), user_id UUID UNIQUE REFERENCES users(id) ON DELETE CASCADE, company_name VARCHAR(255), designation VARCHAR(100), industry VARCHAR(100), created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ); CREATE TABLE IF NOT EXISTS user_roles ( user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE, role_id INTEGER NOT NULL REFERENCES roles(id) ON DELETE CASCADE, assigned_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, PRIMARY KEY (user_id, role_id) ); CREATE TABLE IF NOT EXISTS placement_drives ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), recruiter_id UUID REFERENCES recruiter_profiles(id), title VARCHAR(255), description TEXT, min_cgpa NUMERIC(4,2), eligible_departments TEXT[], package_offered NUMERIC(10,2), location VARCHAR(255), application_deadline TIMESTAMP, status VARCHAR(50) DEFAULT 'DRAFT', is_featured BOOLEAN DEFAULT FALSE, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, employment_type TEXT, max_backlogs INTEGER DEFAULT 0, min_year_of_study INTEGER, drive_start_date TIMESTAMP, drive_end_date TIMESTAMP, created_by UUID REFERENCES users(id), updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ); CREATE TABLE IF NOT EXISTS applications ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), student_id UUID REFERENCES student_profiles(id) ON DELETE CASCADE, drive_id UUID REFERENCES placement_drives(id) ON DELETE CASCADE, status VARCHAR(50) DEFAULT 'APPLIED', opted_out BOOLEAN DEFAULT FALSE, applied_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, UNIQUE (student_id, drive_id) ); CREATE TABLE IF NOT EXISTS drive_required_skills ( drive_id UUID NOT NULL REFERENCES placement_drives(id) ON DELETE CASCADE, skill_id INTEGER NOT NULL REFERENCES skills(id), weight INTEGER DEFAULT 1, PRIMARY KEY (drive_id, skill_id) ); CREATE TABLE IF NOT EXISTS drive_rounds ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), drive_id UUID REFERENCES placement_drives(id) ON DELETE CASCADE, round_name VARCHAR(100), sequence_number INTEGER NOT NULL, round_type VARCHAR(50), scheduled_datetime TIMESTAMP, mode VARCHAR(50), location_or_link TEXT, status VARCHAR(50) DEFAULT 'UPCOMING' ); CREATE TABLE IF NOT EXISTS round_results ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), round_id UUID REFERENCES drive_rounds(id) ON DELETE CASCADE, student_id UUID REFERENCES student_profiles(id) ON DELETE CASCADE, result_status VARCHAR(50), score NUMERIC(6,2), remarks TEXT, evaluated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, UNIQUE (round_id, student_id) ); CREATE TABLE IF NOT EXISTS offers ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), student_id UUID REFERENCES student_profiles(id), drive_id UUID REFERENCES placement_drives(id), offer_date TIMESTAMP, package NUMERIC(10,2), joining_date DATE, offer_status VARCHAR(50) DEFAULT 'ISSUED', created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ); CREATE TABLE IF NOT EXISTS placements ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), student_id UUID REFERENCES student_profiles(id), drive_id UUID REFERENCES placement_drives(id), final_package NUMERIC(10,2), accepted_at TIMESTAMP, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ); CREATE TABLE IF NOT EXISTS resumes ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), student_id UUID REFERENCES student_profiles(id) ON DELETE CASCADE, file_path TEXT NOT NULL, is_active BOOLEAN DEFAULT TRUE, uploaded_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ); CREATE TABLE IF NOT EXISTS resume_scores ( resume_id UUID PRIMARY KEY REFERENCES resumes(id) ON DELETE CASCADE, skill_score NUMERIC(5,2), experience_score NUMERIC(5,2), completeness_score NUMERIC(5,2), final_score NUMERIC(6,2), computed_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ); CREATE TABLE IF NOT EXISTS certifications ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), student_id UUID REFERENCES student_profiles(id) ON DELETE CASCADE, title VARCHAR(255), issuing_body VARCHAR(255), issue_date DATE, certificate_file TEXT, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ); CREATE TABLE IF NOT EXISTS student_marks ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), student_id UUID REFERENCES student_profiles(id) ON DELETE CASCADE, semester INTEGER NOT NULL, sgpa NUMERIC(4,2), memorandum_file TEXT, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ); CREATE TABLE IF NOT EXISTS student_skills ( student_id UUID NOT NULL REFERENCES student_profiles(id) ON DELETE CASCADE, skill_id INTEGER NOT NULL REFERENCES skills(id) ON DELETE CASCADE, proficiency_level INTEGER, PRIMARY KEY (student_id, skill_id), CONSTRAINT student_skills_proficiency_level_check CHECK (proficiency_level >= 1 AND proficiency_level <= 5) ); CREATE TABLE IF NOT EXISTS notifications ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), user_id UUID REFERENCES users(id) ON DELETE CASCADE, type VARCHAR(100), message TEXT, is_read BOOLEAN DEFAULT FALSE, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ); CREATE TABLE IF NOT EXISTS password_resets ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), user_id UUID REFERENCES users(id) ON DELETE CASCADE, reset_token TEXT NOT NULL, expires_at TIMESTAMP NOT NULL, used BOOLEAN DEFAULT FALSE, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ); CREATE TABLE IF NOT EXISTS audit_logs ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), user_id UUID REFERENCES users(id), action VARCHAR(100), entity VARCHAR(100), entity_id UUID, details JSONB, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP );
